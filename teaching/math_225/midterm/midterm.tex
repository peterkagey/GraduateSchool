\question[5] {
  Show that $\det(A)=(1+2x^2)^3$ where \[
    A = \begin{bmatrix}
      1    &-2x     & 2x^2 \\
      2x   & 1-2x^2 & -2x \\
      2x^2 & 2x     & 1
    \end{bmatrix}.
  \]
}
\pagebreak

\question[6] {
  Consider the system of linear equations $Ax=b$, where \[
    A=\begin{bmatrix} 1 &-k & k^2\\ 1 & 0 & k\\ 0 & 1 & -1\end{bmatrix}\!, ~~
    x=\begin{bmatrix}x_1 \\x_2\\x_3\end{bmatrix}\!, ~~ \text{and }
    b=\begin{bmatrix} 0 \\ 0\\ 1\end{bmatrix}\!.
  \]
  For what value(s) of $k$ does this have
  \begin{parts}
      \part a unique solution?
      \part infinitely many solutions?
  \end{parts}
  Justify your answer.
}
\pagebreak

\question[6] {
  Find the (complex) eigenvalues of the matrix \[
    A = \begin{bmatrix}
      -5 & -5 & 0\\
      -8 & 1  & 0\\
      -5 & 3  & 7
      \end{bmatrix}.
  \]
}
\pagebreak

\question[9] {
  Assume that $A$ is skew-symmetric $n\times n$ matrix, that is, $A^T=-A$.

  \begin{parts}
      \part (6 points) Prove that $\operatorname{rank}(A)<n$ if $n$ is odd.
      \part (3 points) Find a $2\times 2$ skew-symmetric matrix $A$ such that $\operatorname{rank}(A)=2$.
  \end{parts}
}
\pagebreak
\question[6] {
  Let $S$ be the subspace of $M_3(\mathbb{R})$ consisting of all $3 \times 3$
  symmetric matrices (matrices such that $A = A^T$).
  Find a basis for $S$.
  What's the dimension of $S$?
}

}
